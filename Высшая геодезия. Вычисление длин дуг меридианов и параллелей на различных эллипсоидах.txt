void merid10000(float a, float B1, float B2, float eks2) {
    using namespace std;
    printf("ДАЛЕЕ БУДЕТ ПРОИЗВЕДЕН РАССЧЕТ НЕОБХОДИМЫХ КОЭФФИЦИЕНТОВ а ДЛЯ ЭЛЛИПСОИДА СО ВВЕДЕННЫМИ ПАРАМЕТРАМИ ШИРОТЫ И ЭКСЦЕНТРИСИТЕТА\n\n");
    float a2sin2B1=sin(2*B1);
    float a4sin4B1=sin(4*B1);
    float a6sin6B1=sin(6*B1);
    float a2sin2B2=sin(2*B2);
    float a4sin4B2=sin(4*B2);
    float a6sin6B2=sin(6*B2);
    float m0=a*(1-eks2);
    float m2=1.5*eks2*m0;
    float m4=1.25*eks2*m2;
    float m6=1.17*eks2*m4;
    float a0=m0+(m2/2)+(0.375*m4);
    float a2=(m2/2)+(m4/2)+(0.46875*m6);
    float a4=(m4/8)+(0.1875*m6);
    float a6=(m6/32);
    float X1=(a0*B1)-((a2/2)*a2sin2B1)+((a4/4)*a4sin4B1)-((a6/6)*a6sin6B1);
    float X2=(a0*B2)-((a2/2)*a2sin2B2)+((a4/4)*a4sin4B2)-((a6/6)*a6sin6B2);
    float DeltaX=X2-X1;
    printf("   ЗНАЧЕНИЕ ДЛИНЫ ДУГИ МЕРИДИАНА С ПОГРЕШНОСТЬЮ НЕ БОЛЕЕ 0,0001М = %0.4f \n\n",DeltaX);
    float DeltaXkm=DeltaX/2000;
    float DeltaXmin=DeltaX/120;
    float DeltaXsec=DeltaXmin/60;
    cout <<"-------------------------------------------------------------------------------------------------------------------------------------------------------\n";
    printf("                ЗНАЧЕНИЕ ДЛИНЫ ДУГИ МЕРИДИАНА ДЛЯ 1 ГРАДУСА В КИЛОМЕТРАХ = %0.0f\n\n",DeltaXkm);
    cout <<"-------------------------------------------------------------------------------------------------------------------------------------------------------\n";
    printf("                ЗНАЧЕНИЕ ДЛИНЫ ДУГИ МЕРИДИАНА ДЛЯ 1 МИНУТЫ В МЕТРАХ = %0.0f\n\n",DeltaXmin);
    cout <<"-------------------------------------------------------------------------------------------------------------------------------------------------------\n";
    printf("                ЗНАЧЕНИЕ ДЛИНЫ ДУГИ МЕРИДИАНА ДЛЯ 1 СЕКУНДЫ В МЕТРАХ = %0.0f\n\n",DeltaXsec);
    cout <<"-------------------------------------------------------------------------------------------------------------------------------------------------------\n";
}
 void merid100 (float c, float V12, float V22) {
     using namespace std;
    printf("ДАЛЕЕ БУДЕТ ПРОИЗВЕДЕН РАССЧЕТ ЗНАЧЕНИЯ ДЛИНЫ ДУГИ МЕРИДИАНА С ПОГРЕШНОСТЬЮ НЕ БОЛЕЕ 0.02М ПО ФОРМУЛЕ СИМПСОНА\n\n");
    printf("                ФОРМУЛА СИМПСОНА ВЫГЛЯДИТ СЛЕДУЮЩИМ ОБРАЗОМ: DeltaX=(DealtaB/6)*(M1+4*Mср+M2)\n\n");
    float M1=(c*V12)/(V12*V12);
    float M2=(c*V22)/(V22*V22);
    float Mcr=(M1+M2)/2;
    float DeltaB=0.035;
    float DeltaX=(DeltaB/6)*(M1+4*Mcr+M2);
    float DeltaXkm=DeltaX/2000;
    float DeltaXmin=DeltaX/120;
    float DeltaXsec=DeltaXmin/60;
    printf("    ЗНАЧЕНИЕ ДУГИ МЕРИДИАНА С ПОГРЕШНОСТЬЮ НЕ БОЛЕЕ 0.02М ПО ФОРМУЛЕ СИМПСОНА=%0.2f\n\n",DeltaX);
    cout <<"-------------------------------------------------------------------------------------------------------------------------------------------------------\n";
    printf("                ЗНАЧЕНИЕ ДЛИНЫ ДУГИ МЕРИДИАНА ДЛЯ 1 ГРАДУСА В КИЛОМЕТРАХ = %0.0f\n\n",DeltaXkm);
    cout <<"-------------------------------------------------------------------------------------------------------------------------------------------------------\n";
    printf("                ЗНАЧЕНИЕ ДЛИНЫ ДУГИ МЕРИДИАНА ДЛЯ 1 МИНУТЫ В МЕТРАХ = %0.0f\n\n",DeltaXmin);
    cout <<"-------------------------------------------------------------------------------------------------------------------------------------------------------\n";
    printf("                ЗНАЧЕНИЕ ДЛИНЫ ДУГИ МЕРИДИАНА ДЛЯ 1 СЕКУНДЫ В МЕТРАХ = %0.0f\n\n",DeltaXsec);
    cout <<"-------------------------------------------------------------------------------------------------------------------------------------------------------\n";
}

void paral(float N1, float N2, float B1, float B2){
    using namespace std;
    printf("ДАЛЕЕ БУДЕТ ПРОИЗВЕДЕН РАССЧЕТ ДЛИНЫ ПАРАЛЛЕЛИ ПРИ РАЗНОСТИ ДОЛГОТ 2 ГРАДУСА НА ШИРОТАХ ПУНКТОВ С В1 И В2 ТЕХ ЖЕ ЭЛЛИПСОИДОВ\n\n");

    printf("    ЗНАЧЕНИЕ ДУГИ ПАРАЛЛЕЛИ ПРИ В1 = %0.4f\n\n",DeltaY1);
    printf("    ЗНАЧЕНИЕ ДУГИ ПАРАЛЛЕЛИ ПРИ В2 = %0.4f\n\n",DeltaY2);
